#dosen Fakultas Teknologi Kelautan (FTK)
def dftk(conn):
	cursor = conn.cursor()
	hasil2 = cursor.execute("select count( distinct dosen) as Dosen from Dataset2 where fakultas = 'Fakultas Teknologi Kelautan (FTK)'")
	while 1:
		hasil2 = cursor.fetchone()
		if not hasil2:
			break
		return hasil2.Dosen
